\section{Evaluation Metrics}\label{sec:method:metrics}

Let $y_n$ and $\hat{y}_n$ denote reference and predicted cluster indices for pulse $n\in\{1,\ldots,N\}$, as in \cref{eq:method:cardinalities,eq:method:dual_embeddings}, instantiated by $(e_n,\hat{e}_n)$ or $(m_n,\hat{m}_n)$.
The contingency is $n_{ij}=\sum_{n=1}^N \indicator[y_n=i,\,\hat{y}_n=j]$ with marginals $a_i=\sum_j n_{ij}$ and $b_j=\sum_i n_{ij}$.

\subsection*{Adjusted Rand index}

The \gls{ari} \parencite{hubert1985comparing} is
\begin{equation}\label{eq:method:ari}
  \ARI
  =
  \frac{
    \displaystyle\sum_{i,j}\binom{n_{ij}}{2}
    -
    \bigl[\sum_i\binom{a_i}{2}\bigr]\bigl[\sum_j\binom{b_j}{2}\bigr]\big/\binom{N}{2}
  }{
    \displaystyle\frac{1}{2}\Bigl[\sum_i\binom{a_i}{2}+\sum_j\binom{b_j}{2}\Bigr]
    -
    \bigl[\sum_i\binom{a_i}{2}\bigr]\bigl[\sum_j\binom{b_j}{2}\bigr]\big/\binom{N}{2}
  }\,,
  \qquad \ARI\le 1\,.
\end{equation}
Perfect agreement up to relabelling gives $\ARI=1$.

\subsection*{Entropies, mutual information, and normalizations}

Shannon entropies of the marginals are
\begin{equation}\label{eq:method:entropy_marginals}
  H(Y)=-\sum_{i:\,a_i>0}\frac{a_i}{N}\log\frac{a_i}{N}\,,
  \qquad
  H(\hat{Y})=-\sum_{j:\,b_j>0}\frac{b_j}{N}\log\frac{b_j}{N}\,.
\end{equation}
The empirical mutual information is \parencite{vinh2010information}
\begin{equation}\label{eq:method:mi}
  \mathrm{MI}(Y;\hat{Y})
  =
  \sum_{i,j:\,n_{ij}>0}\frac{n_{ij}}{N}\,
  \log\frac{N\,n_{ij}}{a_i b_j}\,,
\end{equation}
with $0\log 0=0$.
A \gls{nmi} score is obtained by dividing $\mathrm{MI}$ by a function of $\{H(Y),H(\hat{Y})\}$; we use the arithmetic-mean denominator \parencite{vinh2010information},
\begin{equation}\label{eq:method:nmi}
  \NMI(Y;\hat{Y})
  =
  \frac{2\,\mathrm{MI}(Y;\hat{Y})}{H(Y)+H(\hat{Y})}\,,
  \qquad H(Y)+H(\hat{Y})>0\,,
\end{equation}
and set $\NMI=0$ if $H(Y)+H(\hat{Y})=0$.
Let $\mathbb{E}[\mathrm{MI}]$ be the expectation of $\mathrm{MI}$ under the permutation null that fixes one partition while permuting labels subject to fixed cluster sizes \parencite{vinh2010information}.
The \gls{ami} is
\begin{equation}\label{eq:method:ami}
  \AMI(Y;\hat{Y})
  =
  \frac{\mathrm{MI}(Y;\hat{Y})-\mathbb{E}[\mathrm{MI}]}{\tfrac{1}{2}\bigl(H(Y)+H(\hat{Y})\bigr)-\mathbb{E}[\mathrm{MI}]}\,,
\end{equation}
when the denominator is positive; otherwise $\AMI\equiv 0$.

\subsection*{Clustering accuracy}

Cluster indices are arbitrary up to a bijection $\phi$ on label symbols.
The \gls{acc} is the largest agreement after an optimal relabeling,
\begin{equation}\label{eq:method:acc}
  \ACC
  =
  \frac{1}{N}\max_{\phi \in \Phi}\;\sum_{n=1}^N
  \indicator\bigl[y_n = \phi(\hat{y}_n)\bigr]\,,
\end{equation}
where $\Phi$ is the set of bijections between the active reference labels and the active predicted labels when their counts coincide; if cardinalities differ, the same maximum is computed by the Hungarian (linear-sum assignment) algorithm on the matrix $(n_{ij})$ with cost $-n_{ij}$ \parencite{kuhn1955hungarian}.

\subsection*{V-measure}

With conditional entropies
\begin{equation}\label{eq:method:cond_ent}
  H(Y\mid \hat{Y})=-\sum_{i,j:\,n_{ij}>0}\frac{n_{ij}}{N}\log\frac{n_{ij}}{b_j}\,,
  \qquad
  H(\hat{Y}\mid Y)=-\sum_{i,j:\,n_{ij}>0}\frac{n_{ij}}{N}\log\frac{n_{ij}}{a_i}\,,
\end{equation}
homogeneity and completeness are \parencite{rosenberg2007v}
\begin{equation}\label{eq:method:vmeasure_hc}
  h = 1-\frac{H(Y\mid \hat{Y})}{H(Y)}\,,\qquad
  c = 1-\frac{H(\hat{Y}\mid Y)}{H(\hat{Y})}\,,
\end{equation}
with $h=1$ if $H(Y)=0$ and $c=1$ if $H(\hat{Y})=0$.
The \gls{vmeasure} is
\begin{equation}\label{eq:method:vmeasure}
  \mathrm{V}(Y,\hat{Y})=\frac{2hc}{h+c}\,,\qquad h+c>0\,,
\end{equation}
and $\mathrm{V}=0$ if $h=c=0$.
