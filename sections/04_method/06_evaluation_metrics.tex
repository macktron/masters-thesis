\section{Evaluation Metrics}\label{sec:method:metrics}

Let $y_n$ denote a reference label and $\hat{y}_n$ a predicted cluster index for pulse $n\in\{1,\ldots,N\}$, as in \cref{eq:method:cardinalities,eq:method:dual_embeddings}.
The same formulas apply when $(y_n,\hat{y}_n)$ is instantiated by emitter pairs $(e_n,\hat{e}_n)$ or mode pairs $(m_n,\hat{m}_n)$.
Write $n_{ij}$ for the contingency count with reference class $i$ and predicted cluster $j$, with row and column sums $a_i=\sum_j n_{ij}$ and $b_j=\sum_i n_{ij}$.

\subsection*{Adjusted Rand index}

The \gls{ari} compares clusterings through pair counts and corrects for chance \parencite{hubert1985comparing},
\begin{equation}\label{eq:method:ari}
  \ARI
  =
  \frac{
    \displaystyle\sum_{i,j}\binom{n_{ij}}{2}
    -
    \bigl[\sum_i\binom{a_i}{2}\bigr]\bigl[\sum_j\binom{b_j}{2}\bigr]\big/\binom{N}{2}
  }{
    \displaystyle\frac{1}{2}\Bigl[\sum_i\binom{a_i}{2}+\sum_j\binom{b_j}{2}\Bigr]
    -
    \bigl[\sum_i\binom{a_i}{2}\bigr]\bigl[\sum_j\binom{b_j}{2}\bigr]\big/\binom{N}{2}
  }\,.
\end{equation}
It satisfies $\ARI\le 1$, with $\ARI=1$ for identical partitions up to renaming clusters.

\subsection*{Adjusted mutual information}

View $Y$ and $\hat{Y}$ as the labels of a uniformly sampled pulse index.
Their (empirical) mutual information is
\begin{equation}\label{eq:method:mi}
  \mathrm{MI}(Y;\hat{Y})
  =
  \sum_{i,j}\frac{n_{ij}}{N}\,
  \log\frac{N\,n_{ij}}{a_i b_j}\,,
\end{equation}
with the convention $0\log 0=0$.
Let $\mathbb{E}[\mathrm{MI}]$ denote the expectation of mutual information under the permutation null that fixes one marginal partition while permuting assignments subject to fixed cluster sizes \parencite{vinh2010information}.
The \gls{ami} is
\begin{equation}\label{eq:method:ami}
  \AMI(Y;\hat{Y})
  =
  \frac{\mathrm{MI}(Y;\hat{Y})-\mathbb{E}[\mathrm{MI}]}{\tfrac{1}{2}\bigl(H(Y)+H(\hat{Y})\bigr)-\mathbb{E}[\mathrm{MI}]}\,,
\end{equation}
where $H(\cdot)$ is Shannon entropy of the corresponding label distribution (the denominator uses the arithmetic mean of the marginal entropies, as in the default of common software implementations \parencite{vinh2010information}).

\subsection*{V-measure}

Let $H(Y\mid \hat{Y})$ and $H(\hat{Y}\mid Y)$ be conditional entropies computed from the same contingency table as in \cref{eq:method:mi}.
Homogeneity and completeness are \parencite{rosenberg2007v}
\begin{equation}\label{eq:method:vmeasure_hc}
  h = 1-\frac{H(Y\mid \hat{Y})}{H(Y)}\,,\qquad
  c = 1-\frac{H(\hat{Y}\mid Y)}{H(\hat{Y})}\,,
\end{equation}
with $h=1$ when $H(Y)=0$ and $c=1$ when $H(\hat{Y})=0$.
The \gls{vmeasure} is their harmonic mean,
\begin{equation}\label{eq:method:vmeasure}
  \mathrm{V}(Y,\hat{Y})=\frac{2hc}{h+c}\,,\qquad h+c>0\,,
\end{equation}
and $\mathrm{V}=0$ if $h=c=0$.

Embeddings $\{\vect{z}^{\mathrm{em}}_n\}$ or $\{\vect{z}^{\mathrm{md}}_n\}$ are projected to two dimensions for scatter plots coloured by reference labels, as a qualitative check of local geometry when scores in \cref{eq:method:ari,eq:method:ami,eq:method:vmeasure} are tied across methods.
